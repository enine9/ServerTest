public class Test {
 public static void main(String[] args) {
  byte[] message;
  byte[] Head=new byte[3];
  byte[] Body=new byte[4]; 
  
  message=new byte[Head.length+Body.length];
  
  for (int i = 0; i < message.length; i++) {
   if(i<Head.length){
    message[i]=Head[i];
   }else{
    message[i]=Body[i-Head.length];
   }
  }
 }
}
